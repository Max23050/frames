function removeFromCart(imageSrc) {
    const images = cartItems.querySelectorAll('img');
    // Начинаем поиск с конца списка, чтобы найти последнее добавленное изображение товара
    for (let i = images.length - 1; i >= 0; i--) {

            cartItems.removeChild(images[i]);
            // После удаления одного изображения выходим из цикла

    });
}

Если размеры изображений могут отличаться или вы хотите использовать их реальные размеры для расчета смещения, вам нужно будет дождаться момента загрузки изображения перед добавлением его в DOM и расчетом смещения. Это можно реализовать, добавив обработчик события load к каждому изображению.

Выбор подхода зависит от вашей конкретной задачи и требований к визуальному отображению. Упрощенный подход хорошо работает в большинстве случаев, особенно если размеры изображений стандартизированы или различие между их размерами незначительно.

function updateItemsPosition() {
    const cartItemsContainer = document.querySelector('.content__cart-block');
    const cartItems = cartItemsContainer.querySelectorAll('.cart-item');
    const containerWidth = cartItemsContainer.offsetWidth;
    const totalItemsWidth = Array.from(cartItems).reduce((total, item) => total + item.offsetWidth, 0) + (cartItems.length - 1) * 50; // Сумма ширин всех элементов + смещение
    let leftPosition = (containerWidth - totalItemsWidth) / 2; // Начальное смещение для центрирования
    
    cartItems.forEach((item, index) => {
        if (leftPosition < 0) leftPosition = 0; // Если элементы не умещаются, начинаем с нуля
        item.style.position = 'relative';
        item.style.left = `${leftPosition}px`;
        leftPosition += item.offsetWidth + 50; // Увеличиваем leftPosition на ширину элемента и смещение
    });

    if (totalItemsWidth > containerWidth) {
        cartItemsContainer.style.justifyContent = 'flex-start';
    } else {
        cartItemsContainer.style.justifyContent = 'center';
